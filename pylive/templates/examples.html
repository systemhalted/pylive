class Person(object):
    def __init__(self):
        self._first_name = ""

    @property
    def first_name(self):
        return self._first_name

    @first_name.setter
    def first_name(self, first_name):
        self._first_name = first_name

    @first_name.deleter
    def first_name(self):
        del self._first_name

p = Person()
p.first_name = "Guido"
print p.first_name
